49  // NOTE:without testName, this will not call @Before* and fail w/NPE during surefire
50  @Test(groups = "unit", testName = "SpotInstanceHandlerTest")
51  public class SpotInstanceHandlerTest extends BaseEC2HandlerTest {
52  
53     private DateService dateService;
54  
55     @BeforeTest
56     @Override
57     protected void setUpInjector() {
58        injector = Guice.createInjector(new SaxParserModule(), new AbstractModule() {
59  
60           @Override
61           protected void configure() {
62              bind(String.class).annotatedWith(Region.class).toInstance("us-east-1");
63           }
64  
65        });
66        factory = injector.getInstance(ParseSax.Factory.class);
67        dateService = injector.getInstance(DateService.class);
68        assert dateService != null;
69     }
70  
71     public void testApplyInputStream() {
72  
73        InputStream is = getClass().getResourceAsStream("/request_spot_instances-ebs.xml");
74  
75        SpotInstanceRequest expected = SpotInstanceRequest
76              .builder()
77              .region("us-east-1")
78              .id("sir-228e6406")
79              .spotPrice(0.001f)
80              .type(SpotInstanceRequest.Type.ONE_TIME)
81              .state(SpotInstanceRequest.State.OPEN)
82              .launchSpecification(
83                    LaunchSpecification.builder().imageId("ami-595a0a1c").groupId("default").instanceType("m1.large")
84                          .mapNewVolumeToDevice("/dev/sda1", 1, true)
85                          .mapEBSSnapshotToDevice("/dev/sda2", "snap-1ea27576", 1, true)
86                          .mapEphemeralDeviceToDevice("/dev/sda3", "vre1").monitoringEnabled(false).build())
87              .createTime(new SimpleDateFormatDateService().iso8601DateParse("2011-03-08T03:30:36.000Z"))
88              .productDescription("Linux/UNIX").build();
89        SpotInstanceHandler handler = injector.getInstance(SpotInstanceHandler.class);
90        addDefaultRegionToHandler(handler);
91        SpotInstanceRequest result = factory.create(handler).parse(is);
92        assertEquals(result, expected);
93     }
94  
95     private void addDefaultRegionToHandler(ParseSax.HandlerWithResult<?> handler) {
96        GeneratedHttpRequest<?> request = createMock(GeneratedHttpRequest.class);
97        expect(request.getArgs()).andReturn(ImmutableList.<Object> of()).atLeastOnce();
98        replay(request);
99        handler.setContext(request);
100    }
