57     @Override
58     protected Predicate<OsFamilyVersion64Bit> defineUnsupportedOperatingSystems() {
59        return Predicates.not(new Predicate<OsFamilyVersion64Bit>() {
60  
61           @Override
62           public boolean apply(OsFamilyVersion64Bit input) {
63              switch (input.family) {
64                 case UBUNTU:
65                    return true;
66                 case DEBIAN:
67                    return true;
68                 case CENTOS:
69                    return input.version.matches("5.[42]") || input.version.equals("");
70                 case WINDOWS:
71                    return input.version.matches("200[38]") || input.version.equals("");
72                 default:
73                    return false;
74              }
75           }
76  
77        });
78     }

90        assertEquals(template.getImage().getUserMetadata().get("rootDeviceType"), "instance-store");

92        assertEquals(getCores(template.getHardware()), 1.0d);

160    @Test
161    public void testTemplateBuilderWithNoOwnersParsesImageOnDemand() throws IOException {
162       ComputeServiceContext context = null;
163       try {
164          Properties overrides = setupProperties();
165          // set owners to nothing
166          overrides.setProperty(EC2Constants.PROPERTY_EC2_AMI_OWNERS, "");
167 
168          context = new ComputeServiceContextFactory().createContext(provider, ImmutableSet
169                   .<Module> of(new Log4JLoggingModule()), overrides);
170 
171          assertEquals(context.getComputeService().listImages().size(), 0);
172 
173          Template template = context.getComputeService().templateBuilder().imageId("us-east-1/ami-ccb35ea5").build();
174          System.out.println(template.getHardware());
175          assert (template.getImage().getProviderId().startsWith("ami-")) : template;
176          assertEquals(template.getImage().getOperatingSystem().getVersion(), "5.4");
177          assertEquals(template.getImage().getOperatingSystem().is64Bit(), true);
178          assertEquals(template.getImage().getOperatingSystem().getFamily(), OsFamily.CENTOS);
179          assertEquals(template.getImage().getVersion(), "4.4.10");
180          assertEquals(template.getImage().getUserMetadata().get("rootDeviceType"), "instance-store");
181          assertEquals(template.getLocation().getId(), "us-east-1");
182          assertEquals(getCores(template.getHardware()), 2.0d);
183          assertEquals(template.getHardware().getId(), "m1.large"); // because it
184          // is 64bit
185 
186          // ensure we cache the new image for next time
187          assertEquals(context.getComputeService().listImages().size(), 1);
188 
189       } finally {
190          if (context != null)
191             context.close();
192       }
193    }
