246    @Test(timeOut = 120000)
247    public void testGetNodeMetadata() throws Exception {
248       Set<NodeMetadata> nodes = Sets.newHashSet(getDefaultNodeImmediately(group));
249       awaitForPublicAddressAssigned(nodes.iterator().next().getId());
250       Map<String, ? extends NodeMetadata> metadataMap = newLinkedHashMap(uniqueIndex(filter(computeService
251             .listNodesDetailsMatching(all()), and(inGroup(group), not(TERMINATED))),
252             new Function<NodeMetadata, String>() {
253 
254                @Override
255                public String apply(NodeMetadata from) {
256                   return from.getId();
257                }
258 
259             }));
260       for (NodeMetadata node : nodes) {
261          metadataMap.remove(node.getId());
262          NodeMetadata nodeMetadata = computeService.getNodeMetadata(node.getId());
263          assertEquals(parseGroupFromName(nodeMetadata.getName()), group);
264          assertEquals(nodeMetadata.getProviderId(), node.getProviderId());
265          assertEquals(nodeMetadata.getGroup(), node.getGroup());
266          assertLocationSameOrChild(nodeMetadata.getLocation(), getDefaultTemplateBuilder().build().getLocation());
267          assertEquals(nodeMetadata.getImageId(), getDefaultTemplateBuilder().build().getImage().getId());
268 //         checkOsMatchesTemplate(metadata);
269          assertEquals(nodeMetadata.getState(), NodeState.RUNNING);
270          // due to DHCP the addresses can actually change in-between runs.
271          assertTrue(nodeMetadata.getPrivateAddresses().size() > 0);
272          assertTrue(nodeMetadata.getPublicAddresses().size() > 0);
273       }
274       assertNodeZero(metadataMap.values(), nodes);
275    }

348    @Test(timeOut = 60000)
349    public void testGetAssignableLocations() throws Exception {
350       for (Location location : computeService.listAssignableLocations()) {
351          System.err.printf("location %s%n", location);
352          assert location.getId() != null : location;
353          assert location != location.getParent() : location;
354          assert location.getScope() != null : location;
355          switch (location.getScope()) {
356             case PROVIDER:
357                assertProvider(location);
358                break;
359             case REGION:
360                assertProvider(location.getParent());
361                break;
362             case ZONE:
363                Location provider = location.getParent().getParent();
364                // zone can be a direct descendant of provider
365                if (provider == null)
366                   provider = location.getParent();
367                assertProvider(provider);
368                break;
369             case HOST:
370                Location provider2 = location.getParent().getParent().getParent();
371                // zone can be a direct descendant of provider
372                if (provider2 == null)
373                   provider2 = location.getParent().getParent();
374                assertProvider(provider2);
375                break;
376          }
377       }
378    }

435    protected void doCheckJavaIsInstalledViaSsh(NodeMetadata node) throws IOException {
436 
437       SshClient ssh = context.utils().sshForNode().apply(node);
438       try {
439          ssh.connect();
440          ExecResponse hello = ssh.exec("echo hello");
441          assertEquals(hello.getOutput().trim(), "hello");
442          ExecResponse exec = ssh.exec("java -version");
443          assert exec.getError().indexOf("1.6") != -1 || exec.getOutput().indexOf("1.6") != -1 : exec + "\n"
444                + ssh.exec("cat /tmp/bootstrap/stdout.log /tmp/bootstrap/stderr.log");
445       } finally {
446          if (ssh != null)
447             ssh.disconnect();
448       }
