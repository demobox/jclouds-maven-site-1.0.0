197    private void blockUntilPublicAddress(int serverId) throws InterruptedException {
198       while (client.getServer(serverId).getAddresses().getPublicAddresses().isEmpty()) {
199          System.out.println("Awaiting public address");
200          Thread.sleep(1000);
201       }
202    }

204    private void blockUntilServerActive(int serverId) throws InterruptedException {
205       Server currentDetails;
206       for (currentDetails = client.getServer(serverId); currentDetails.getStatus() != ServerStatus.ACTIVE; currentDetails = client
207             .getServer(serverId)) {
208          System.out.printf("blocking on status active%n%s%n", currentDetails);
209          Thread.sleep(5 * 1000);
210       }
211    }
212 
213    private void blockUntilServerVerifyResize(int serverId) throws InterruptedException {
214       Server currentDetails;
215       for (currentDetails = client.getServer(serverId); currentDetails.getStatus() != ServerStatus.VERIFY_RESIZE; currentDetails = client
216             .getServer(serverId)) {
217          System.out.printf("blocking on status verify resize%n%s%n", currentDetails);
218          Thread.sleep(5 * 1000);
219       }
220    }
