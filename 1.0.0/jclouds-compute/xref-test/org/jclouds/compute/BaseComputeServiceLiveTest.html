218       String group = this.group + "r";
219       try {
220          client.destroyNodesMatching(inGroup(group));
221       } catch (Exception e) {
222 
223       }
224 
225       TemplateOptions options = client.templateOptions().blockOnPort(22, 120);
226       try {
227          Set<? extends NodeMetadata> nodes = client.createNodesInGroup(group, 1, options);
228          Credentials good = nodes.iterator().next().getCredentials();
229          assert good.identity != null : nodes;
230          assert good.credential != null : nodes;
231 
232          OperatingSystem os = get(nodes, 0).getOperatingSystem();
233          try {
234             Map<? extends NodeMetadata, ExecResponse> responses = runScriptWithCreds(group, os, new Credentials(
235                      good.identity, "romeo"));
236             assert false : "shouldn't pass with a bad password\n" + responses;
237          } catch (RunScriptOnNodesException e) {
238             assert getRootCause(e).getMessage().contains("Auth fail") : e;
239          }
240 
241          for (Entry<? extends NodeMetadata, ExecResponse> response : client.runScriptOnNodesMatching(
242                   runningInGroup(group), Statements.exec("echo hello"),
243                   overrideCredentialsWith(good).wrapInInitScript(false).runAsRoot(false)).entrySet())
244             assert response.getValue().getOutput().trim().equals("hello") : response.getKey() + ": "
245                      + response.getValue();
246 
247          // test single-node execution
248          ExecResponse response = client.runScriptOnNode(get(nodes, 0).getId(), "echo hello", wrapInInitScript(false)
249                   .runAsRoot(false));
250          assert response.getOutput().trim().equals("hello") : get(nodes, 0).getId() + ": " + response;
251 
252          runScriptWithCreds(group, os, good);
253 
254          checkNodes(nodes, group);
255 
256       } finally {
257          client.destroyNodesMatching(inGroup(group));
258       }
259    }

557                   provider2 = location.getParent().getParent();
