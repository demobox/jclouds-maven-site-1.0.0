57  @Test(groups = "live", sequential = true)
58  public class AvailabilityZoneAndRegionClientLiveTest {

60     private AvailabilityZoneAndRegionClient client;
61     private RestContext<EC2Client, EC2AsyncClient> context;
62     protected String provider = "ec2";
63     protected String identity;
64     protected String credential;
65     protected String endpoint;
66     protected String apiversion;

68     @BeforeClass
69     protected void setupCredentials() {
70        identity = checkNotNull(System.getProperty("test." + provider + ".identity"), "test." + provider + ".identity");
71        credential = System.getProperty("test." + provider + ".credential");
72        endpoint = System.getProperty("test." + provider + ".endpoint");
73        apiversion = System.getProperty("test." + provider + ".apiversion");
74     }
75  
76     protected Properties setupProperties() {
77        Properties overrides = new Properties();
78        overrides.setProperty(Constants.PROPERTY_TRUST_ALL_CERTS, "true");
79        overrides.setProperty(Constants.PROPERTY_RELAX_HOSTNAME, "true");
80        overrides.setProperty(provider + ".identity", identity);
81        if (credential != null)
82           overrides.setProperty(provider + ".credential", credential);
83        if (endpoint != null)
84           overrides.setProperty(provider + ".endpoint", endpoint);
85        if (apiversion != null)
86           overrides.setProperty(provider + ".apiversion", apiversion);
87        return overrides;
88     }
89  
90     @BeforeGroups(groups = { "live" })
91     public void setupClient() {
92        setupCredentials();
93        Properties overrides = setupProperties();
94        context = new ComputeServiceContextFactory().createContext(provider,
95              ImmutableSet.<Module> of(new Log4JLoggingModule()), overrides).getProviderSpecificContext();
96        client = context.getApi().getAvailabilityZoneAndRegionServices();
97     }
98  
99     public void testDescribeAvailabilityZones() {
100       for (String region : Lists.newArrayList(null, Region.EU_WEST_1, Region.US_EAST_1, Region.US_WEST_1,
101             Region.AP_SOUTHEAST_1)) {
102          Set<AvailabilityZoneInfo> allResults = client.describeAvailabilityZonesInRegion(region);
103          assertNotNull(allResults);
104          assert allResults.size() >= 2 : allResults.size();
105          Iterator<AvailabilityZoneInfo> iterator = allResults.iterator();
106          String id1 = iterator.next().getZone();
107          String id2 = iterator.next().getZone();
108          Set<AvailabilityZoneInfo> twoResults = client.describeAvailabilityZonesInRegion(region,
109                availabilityZones(id1, id2));
110          assertNotNull(twoResults);
111          assertEquals(twoResults.size(), 2);
112          iterator = allResults.iterator();
113          assertEquals(iterator.next().getZone(), id1);
114          assertEquals(iterator.next().getZone(), id2);
115       }
116    }
117 
118    public void testDescribeRegions() {
119       SortedMap<String, URI> allResults = Maps.newTreeMap();
120       allResults.putAll(client.describeRegions());
121       assertNotNull(allResults);
122       assert allResults.size() >= 2 : allResults.size();
123       Iterator<Entry<String, URI>> iterator = allResults.entrySet().iterator();
124       String r1 = iterator.next().getKey();
125       String r2 = iterator.next().getKey();
126       SortedMap<String, URI> twoResults = Maps.newTreeMap();
127       twoResults.putAll(client.describeRegions(regions(r1, r2)));
128       assertNotNull(twoResults);
129       assertEquals(twoResults.size(), 2);
130       iterator = twoResults.entrySet().iterator();
131       assertEquals(iterator.next().getKey(), r1);
132       assertEquals(iterator.next().getKey(), r2);
133    }
