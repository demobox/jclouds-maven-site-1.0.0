392    protected void doCheckPass(String address) throws IOException {
393       IPSocket socket = new IPSocket(address, 22);
394 
395       System.out.printf("%d: %s awaiting ssh service to start%n", System.currentTimeMillis(), socket);
396       assert socketTester.apply(socket);
397       System.out.printf("%d: %s ssh service started%n", System.currentTimeMillis(), socket);
398 
399       SshClient connection = getConnectionFor(socket);
400       try {
401          connection.connect();
402          System.out.printf("%d: %s ssh connection made%n", System.currentTimeMillis(), socket);
403          System.out.println(connection.exec("df -h"));
404          System.out.println(connection.exec("ls -al /dev/sd*"));
405          System.out.println(connection.exec("echo '$Ep455l0ud!2'|sudo -S fdisk -l"));
406       } finally {
407          if (connection != null)
408             connection.disconnect();
409       }
410    }
